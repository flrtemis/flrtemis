    """
    Probabilistic Typo Estimation Model
    
    Theorem: Estimated Typos = ⌊Message Length * Typo Rate⌋
    
    Args:
        message_length (int): Total character count

        int: Discrete typo count

def estimate_typos(message_length, typing_speed_factor, device_difficulty_factor, context_complexity_factor, individual_skill_factor):
    """
    Estimates the number of typos in a message based on several factors.

    Args:
        message_length: Length of the message (e.g., number of words or characters).
        typing_speed_factor: Higher value for faster typing (e.g., 1 for slow, 3 for fast).
        device_difficulty_factor: Higher value for more difficult devices (e.g., 1 for desktop, 3 for small phone).
        context_complexity_factor: Higher value for complex words, hurried typing, multitasking (e.g., 1 for simple, 3 for complex).
        individual_skill_factor: Normalizing factor representing individual typing proficiency (higher value for better skills).

    Returns:
        The estimated number of typos.
        Returns a message if there are invalid inputs.
    """

    if not all(isinstance(x, (int, float)) and x >= 0 for x in [message_length, typing_speed_factor, device_difficulty_factor, context_complexity_factor, individual_skill_factor]):
        return "All input values must be non-negative numbers."
    if individual_skill_factor == 0:
        return "Individual skill factor cannot be zero (division by zero)."
    

    estimated_typos = (message_length * (typing_speed_factor + device_difficulty_factor + context_complexity_factor)) / individual_skill_factor
    return estimated_typos

        with open(self.memory_file, "w") as file:

    def send_message(self):
        user_input = self.input_field.get().strip()
        if user_input:
            self.display_message("You", user_input)
            self.input_field.delete(0, tk.END)

            sage_response = self.generate_response(user_input)

            # Get typo rate from slider value (0-100) and convert to 0.0-1.0
            typo_rate = self.typo_slider.get() / 100 
            self.typewriter_display("Sage", sage_response, typo_rate=typo_rate) 

            self.log_conversation(user_input, sage_response)

    def generate_response(self, user_input):
        if "hello" in user_input.lower():
            return "Hello! How can I assist you today?"
        if "remember" in user_input.lower():
            return "I always try to remember things for you!"
        return f"I'm still learning, but I appreciate your input: '{user_input}'"

    def display_message(self, sender, message):
        self.chat_display.config(state="normal")
        self.chat_display.insert(tk.END, f"{sender}: {message}\n")
        self.chat_display.config(state="disabled")
        self.chat_display.see(tk.END)

    def typewriter_display(self, sender, message, delay=70, typo_probability=0.1, typo_rate):
        """Display Sage's response with a typewriter effect and typos.

        Args:
            sender (str): The name of the sender ("Sage").
            message (str): The message to display.
            delay (int): The delay (in milliseconds) between each character.
            typo_probability (float): The probability (0.0-1.0) of a typo occurring at a selected index.
            typo_rate (float): The base rate of typos per character (0.0-1.0). This directly controls the number of typos.
        """
        estimated_typos = estimate_typos(len(message), typo_rate)
        typo_count = min(int(estimated_typos), len(message) // 3)
        typo_indices = random.sample(range(len(message)), typo_count) if len(message) > 0 and typo_count > 0 else []

        def display_char(index=0):
            if index == 0:
                self.chat_display.config(state="normal")
                self.chat_display.insert(tk.END, f"{sender}: ")

            if index < len(message):
                char_to_display = message[index]
                if index in typo_indices:
                    if random.random() < typo_probability:
                        possible_typos = "qwertyuiopasdfghjklzxcvbnm"
                        typo_char = random.choice(possible_typos)
                        self.chat_display.insert(tk.END, typo_char)
                        self.chat_display.see(tk.END)
                        self.chat_display.after(delay, lambda: self.correct_typo(index, char_to_display))
                    else:
                        self.chat_display.insert(tk.END, char_to_display)
                else:
                    self.chat_display.insert(tk.END, char_to_display)

                self.chat_display.see(tk.END)
                self.chat_display.after(delay, display_char, index + 1)
            else:
                self.chat_display.insert(tk.END, "\n")
                self.chat_display.config(state="disabled")

        display_char()

    def correct_typo(self, typo_index, correct_char):
        """Corrects a typo in the chat display."""
        self.chat_display.delete(f"{typo_index + len(f'Sage: ')}", f"{typo_index + len(f'Sage: ')+1}")
        self.chat_display.insert(f"{typo_index + len(f'Sage: ')}", correct_char)
        self.chat_display.see(tk.END)

    def log_conversation(self, user_input, bot_response):
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.conversation_history.append({"timestamp": timestamp, "user": user_input, "bot": bot_response})
        self.save_memory()
